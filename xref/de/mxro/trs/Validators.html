1   package de.mxro.trs;
2   
3   import de.mxro.trs.internal.OffTableRobot;
4   import de.mxro.trs.internal.PlacedRobot;
5   
6   /**
7    * <p>
8    * Allows creating implementations of {@link RobotValidator}.
9    * <p>
10   * Currently there is only one implementation for asserting that the robot stays
11   * within the bounds of a rectangular table. However, it is easy to add more
12   * sophisticated validators, for instance for introducing obstacles to the
13   * field.
14   *
15   */
16  public class Validators {
17  
18  	/**
19  	 * <p>
20  	 * Asserts that the toy robot stays within a rectangular table of the provided
21  	 * dimensions.
22  	 * 
23  	 * @param tableWidth
24  	 *            The number of horizontal fields.
25  	 * @param tableHeight
26  	 *            The number of vertical fields.
27  	 * @return A new instance of {@link RobotValidator}
28  	 */
29  	public static RobotValidator onTable(int tableWidth, int tableHeight) {
30  		if (tableWidth <= 0 || tableHeight <= 0) {
31  			throw new IllegalArgumentException("Invalid table dimensions: " + tableWidth + ", " + tableHeight);
32  		}
33  
34  		return robot -> {
35  			// A robot off the table is never on the table...
36  			if (robot instanceof OffTableRobot) {
37  				return false;
38  			}
39  
40  			// A placed robot must stay within the bounds.
41  			if (robot instanceof PlacedRobot) {
42  				PlacedRobot placedRobot = (PlacedRobot) robot;
43  				return placedRobot.x >= 0 && placedRobot.y >= 0 && placedRobot.x <= tableWidth
44  						&& placedRobot.y <= tableHeight;
45  			}
46  
47  			// Currently there are no more implementations of robot than the two listed
48  			// above.
49  			// The following exception is to aid future developers in debugging issues if
50  			// additional implementations of robot are introduced.
51  			throw new IllegalStateException("Unsupported subclass of " + Robot.class + ": " + robot.getClass());
52  		};
53  	}
54  
55  }
