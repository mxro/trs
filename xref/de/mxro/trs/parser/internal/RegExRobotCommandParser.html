3   import java.util.Arrays;
4   import java.util.List;
5   import java.util.stream.Collectors;
6   import java.util.stream.Stream;
7   
8   import de.mxro.trs.Command;
9   import de.mxro.trs.ParseException;
10  import de.mxro.trs.commands.MoveCommand;
11  import de.mxro.trs.commands.ReportCommand;
12  import de.mxro.trs.commands.TurnLeftCommand;
13  import de.mxro.trs.commands.TurnRightCommand;
14  import de.mxro.trs.parser.RobotCommandsParser;
15  import de.mxro.trs.parser.internal.pattern.PlaceRegexPattern;
16  import de.mxro.trs.parser.internal.pattern.SimpleCommandRegExPattern;
17  
18  /**
19   * <p>
20   * Provides a very simple implementation to parse text containing commands for
21   * the robot.
22   * <p>
23   * This implementation is based on applying simple regular expressions against
24   * input lines.
25   *
26   */
27  public class RegExRobotCommandParser implements RobotCommandsParser {
28  
29  	private final List<RegExPattern> patterns;
30  
31  	@Override
32  	public List<Command> parse(String input) {
33  
34  		String[] lines = input.split("\n");
35  
36  		// parse each line of the input
37  		return Stream.of(lines).map(line -> {
38  			// determine the pattern which matches this line
39  			List<RegExPattern> matchingPatterns = patterns.stream().filter(pattern -> pattern.match(line))
40  					.collect(Collectors.toList());
41  
42  			// expects exactly one pattern to match
43  			if (matchingPatterns.size() != 1) {
44  				throw new ParseException("Invalid line in input: '" + line + "'");
45  			}
46  
47  			return matchingPatterns.get(0).parseCommand(line);
48  
49  		}).collect(Collectors.toList());
50  	}
51  
52  	public RegExRobotCommandParser() {
53  		super();
54  		// Create the pattern instances used to parse the text.
55  		// This list can easily be easily be extended when new commands are required.
56  		// However, if the complexity of the grammar increases, it would be recommended
57  		// to use a more sophisticated approach for parsing the input such as using 
58  		// ANTLR.
59  		this.patterns = Arrays.asList(new SimpleCommandRegExPattern("^MOVE$", new MoveCommand()),
60  				new SimpleCommandRegExPattern("^LEFT$", new TurnLeftCommand()),
61  				new SimpleCommandRegExPattern("^RIGHT$", new TurnRightCommand()),
62  				new SimpleCommandRegExPattern("^REPORT$", new ReportCommand()), new PlaceRegexPattern());
63  	}
64  
65  }
