11  /**
12   * <p>Parses a {@link PlaceCommand} command.
13   */
14  public class PlaceRegexPattern extends RegExPattern {
15  
16  	@Override
17  	public Command parseCommand(String line) {
18  		 Matcher m = pattern.matcher(line);
19  		 m.find();
20  			 
21  		 try {
22  			 return new PlaceCommand(Integer.valueOf(m.group(1)), Integer.valueOf(m.group(2)), Direction.fromText(m.group(3)));
23  		 } catch (NumberFormatException ex) {
24  			 throw new ParseException("Invalid PLACE declaration: "+line, ex);
25  		 }
26  	}
27  
28  	public PlaceRegexPattern() {
29  		super("^PLACE ([^,]*),([^,]*),(.*)$");
30  		
31  	}
32  
33  }
