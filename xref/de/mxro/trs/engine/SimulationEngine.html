1   package de.mxro.trs.engine;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.stream.Stream;
6   
7   import de.mxro.trs.Command;
8   import de.mxro.trs.Robot;
9   import de.mxro.trs.RobotValidator;
10  import de.mxro.trs.internal.OffTableRobot;
11  
12  /**
13   * <p>
14   * Performs simulations of robot movements on the table.
15   *
16   */
17  public class SimulationEngine {
18  
19  	/**
20  	 * <p>
21  	 * Runs a simulation based on a sequence of commands and validators defining the
22  	 * constraints for the commands.
23  	 * 
24  	 * @param commands
25  	 *            The list of commands to be executed. Only valid commands will be
26  	 *            executed.
27  	 * @param validators
28  	 *            The validators commands are to be validated against.
29  	 * @return A list of states for the Robot. The last item in the list represents
30  	 *         the state of the robot at the end of the simulation.
31  	 */
32  	public static List<Robot> run(List<Command> commands, List<RobotValidator> validators) {
33  
34  		List<Robot> results = new ArrayList<Robot>(commands.size());
35  		// Simulation always starts with the robot being off the table.
36  		Robot robot = new OffTableRobot();
37  		// Looping through the commands rather than using a more 'functional' approach
38  		// since this problem itself nicely to an iterative solution (alternatively
39  		// recursion could be applied, of course).
40  		for (Command command : commands) {
41  			// Determine the new state of the robot after a command is applied.
42  			final Robot newState = command.apply(robot);
43  
44  			// Assure the new state is valid. Only then record the new state. Otherwise
45  			// record the previous state of the robot.
46  			Stream<Boolean> validationResults = validators.stream().map(v -> v.validate(newState));
47  			Stream<Boolean> notValid = validationResults.filter(b -> !b);
48  			if (notValid.findAny().isPresent() == false) {
49  				robot = newState;
50  			}
51  			results.add(robot);
52  
53  		}
54  		return results;
55  	}
56  
57  }
