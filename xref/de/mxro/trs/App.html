10  import de.mxro.trs.engine.RenderingEngine;
11  import de.mxro.trs.engine.SimulationEngine;
12  import de.mxro.trs.parser.RobotCommandParsers;
13  
14  /**
15   * <p>The facade class to interact with this library/application.
16   *
17   */
18  public class App {
19  	
20  	public static void main(String[] args) {
21  		if (args.length != 1 || !new File(args[0]).exists()) {
22  			System.err.println("This application expects one parameter with a valid file path.");
23  		}
24  		
25  		// parsing inputs and performing simulation
26  		String results = run(new File(args[0]), Arrays.asList(Validators.onTable(5, 5)));
27  		
28  		// printing results to stdout
29  		System.out.println(results);
30  	}
31  	
32  	
33  	private static String run(File commands, List<RobotValidator> validators) {
34  		
35  		// read file contents into string
36  		String input;
37  		try {
38  			input = new String(Files.readAllBytes(commands.toPath()), "UTF-8");
39  			
40  		} catch (IOException e) {
41  			throw new RuntimeException("Cannot read commands from file: "+commands, e);
42  		}
43  		
44  		// parse commands from string
45  		List<Command> commandsList = RobotCommandParsers.regex().parse(input);
46  		
47  		// run simulation and render the report for the simulation
48  		return RenderingEngine.renderReport(commandsList, SimulationEngine.run(commandsList, validators));
49  	}

51  }
